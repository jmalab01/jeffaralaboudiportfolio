--17. In 2003, were more race bikes built from carbon or titanium (based on the down tube)?
--As output you may show the number of bikes for both materials. Use OrderDate.
--Material	CountOfSerialNumber 

SELECT TubeMaterial.Material, COUNT(Bicycle.SerialNumber) AS CountOfSerialNumber
FROM BIKE..Bicycle
JOIN BIKE..BikeTubes ON Bicycle.SerialNumber = BikeTubes.SerialNumber
JOIN BIKE..TubeMaterial ON BikeTubes.TubeID = TubeMaterial.TubeID
WHERE Bicycle.ModelType = 'Race'
  AND TubeMaterial.Description = 'Down Tube'
  AND YEAR(Bicycle.OrderDate) = 2003
  AND (TubeMaterial.Material = 'Carbon' OR TubeMaterial.Material = 'Titanium')
GROUP BY TubeMaterial.Material;
 
--18. What is the average price paid for the 2001 Shimano XTR rear derailleurs?
--AvgOfPricePaid

SELECT AVG(PurchaseItem.PricePaid) AS AvgOfPricePaid
FROM BIKE..PurchaseItem
JOIN BIKE..Component ON PurchaseItem.ComponentID = Component.ComponentID
JOIN BIKE..Manufacturer ON Component.ManufacturerID = Manufacturer.ManufacturerID
WHERE Component.ProductNumber = 'XTR' 
  AND Manufacturer.ManufacturerName = 'Shimano'
  AND Component.Category = 'Rear Derailleur'
  AND YEAR(PurchaseItem.OrderDate) = 2001;
 
--19. What is the average top tube length for a 54 cm (frame size) road bike built in 1999?
--AvgOfTopTube

FROM BIKE..Bicycle
JOIN BIKE..ModelSize ON Bicycle.ModelType = ModelSize.ModelType
WHERE Bicycle.FrameSize = 54
  AND Bicycle.ModelType = 'Road'
  AND YEAR(Bicycle.OrderDate) = 1999;
 
--20. On average, which costs (list price) more: road tires or mountain bike tires?
--Road	AvgOfListPrice

SELECT Component.Road, AVG(Component.ListPrice) AS AvgOfListPrice
FROM BIKE..Component
WHERE Component.Category = 'Tire'
GROUP BY Component.Road;
 
--21. In May 2003, which employees sold road bikes that they also painted?
--EmployeeID	LastName

SELECT Employee.EmployeeID, Employee.LastName
FROM BIKE..Employee
JOIN BIKE..Bicycle ON Employee.EmployeeID = Bicycle.EmployeeID
WHERE Bicycle.ModelType = 'Road'
  AND Bicycle.Painter = Employee.EmployeeID
  AND MONTH(Bicycle.OrderDate) = 5
  AND YEAR(Bicycle.OrderDate) = 2003;
 
--22. In 2002, was the Old English letter style more popular with some paint jobs?
--PaintID	ColorName	Number of Bikes Painted

SELECT Paint.PaintID, Paint.ColorName, COUNT(Bicycle.SerialNumber) AS NumberOfBikesPainted
FROM BIKE..Bicycle
JOIN BIKE..Paint ON Bicycle.PaintID = Paint.PaintID
WHERE Bicycle.LetterStyleID = 'Old English'
  AND YEAR(Bicycle.OrderDate) = 2002
GROUP BY Paint.PaintID, Paint.ColorName
ORDER BY NumberOfBikesPainted DESC;
 
--23. Which race bikes in 2003 sold for more than the average price of race bikes in 2002?
--SerialNumber	ModelType	OrderDate	SalePrice

WITH AvgRacePrice2002 AS (
  SELECT AVG(SalePrice) AS AvgPrice
  FROM BIKE..Bicycle
  WHERE ModelType = 'Race' AND YEAR(OrderDate) = 2002
)
SELECT Bicycle.SerialNumber, Bicycle.ModelType, Bicycle.OrderDate, Bicycle.SalePrice
FROM BIKE..Bicycle, AvgRacePrice2002
WHERE Bicycle.ModelType = 'Race'
  AND YEAR(Bicycle.OrderDate) = 2003
  AND Bicycle.SalePrice > AvgRacePrice2002.AvgPrice;
 
 
--24. Which component that had no sales (installations) in 2004 has the highest inventory value (cost basis)?
--ManufacturerName	ProductNumber	Category	Value	ComponentID

SELECT TOP 1 Component.ManufacturerID, Manufacturer.ManufacturerName, Component.ProductNumber, 
       Component.Category, (Component.EstimatedCost * Component.QuantityOnHand) AS Value, Component.ComponentID
FROM BIKE..Component
JOIN BIKE..Manufacturer ON Component.ManufacturerID = Manufacturer.ManufacturerID
LEFT JOIN BIKE..BikeParts ON Component.ComponentID = BikeParts.ComponentID
WHERE (BikeParts.DateInstalled IS NULL OR YEAR(BikeParts.DateInstalled) != 2004)
ORDER BY Value DESC;
 
--25. Create a vendor contacts list of all manufacturers and retail stores in California.
--Include only the columns for VendorName and Phone. 
--The retail stores should only include stores that participated in the sale of at least one bicycle in 2004
--Store Name Or Manufacturer Name	Phone

SELECT Manufacturer.ManufacturerName AS VendorName, Manufacturer.Phone
FROM BIKE..Manufacturer
JOIN BIKE..City ON Manufacturer.CityID = City.CityID
WHERE City.State = 'CA'
UNION
SELECT RetailStore.StoreName AS VendorName, RetailStore.Phone
FROM BIKE..RetailStore
JOIN BIKE..City ON RetailStore.CityID = City.CityID
JOIN BIKE..Bicycle ON RetailStore.StoreID = Bicycle.StoreID
WHERE City.State = 'CA' AND YEAR(Bicycle.OrderDate) = 2004;
 
--26. List all of the employees who report to Venetiaan.
--LastName	EmployeeID	LastName	FirstName	Title

SELECT Employee.EmployeeID, Employee.LastName, Employee.FirstName, Employee.Title
FROM BIKE..Employee
WHERE Employee.CurrentManager = (
  SELECT EmployeeID
  FROM BIKE..Employee
  WHERE LastName = 'Venetiaan'
);
 
--27. List the components where the company purchased at least 25 percent more units than it used through June 30, 2000.
--ComponentID	ManufacturerName	ProductNumber	Category	TotalReceived	TotalUsed	NetGain	NetPct	ListPrice

SELECT Component.ComponentID, Manufacturer.ManufacturerName, Component.ProductNumber, Component.Category,
       SUM(PurchaseItem.Quantity) AS TotalReceived, SUM(BikeParts.Quantity) AS TotalUsed,
       (SUM(PurchaseItem.Quantity) - SUM(BikeParts.Quantity)) AS NetGain,
       ((SUM(PurchaseItem.Quantity) - SUM(BikeParts.Quantity)) * 100.0 / SUM(BikeParts.Quantity)) AS NetPct,
       Component.ListPrice
FROM BIKE..Component
JOIN BIKE..Manufacturer ON Component.ManufacturerID = Manufacturer.ManufacturerID
JOIN BIKE..PurchaseItem ON Component.ComponentID = PurchaseItem.ComponentID
LEFT JOIN BIKE..BikeParts ON Component.ComponentID = BikeParts.ComponentID
WHERE PurchaseItem.OrderDate <= '2000-06-30'
  AND (BikeParts.DateInstalled IS NULL OR BikeParts.DateInstalled <= '2000-06-30')
GROUP BY Component.ComponentID, Manufacturer.ManufacturerName, Component.ProductNumber, Component.Category, Component.ListPrice
HAVING NetPct >= 25;
 
--28. In which years did the average build time for the year exceed the overall average build time for all years? 
--The build time is the difference between order date and ship date.
--Use the difference between OrderDate and ShipDate.
--Year	BuildTime

WITH YearlyBuildTime AS (
  SELECT YEAR(OrderDate) AS Year, AVG(DATEDIFF(day, OrderDate, ShipDate)) AS AvgBuildTime
  FROM BIKE..Bicycle
  GROUP BY YEAR(OrderDate)
),
OverallAvgBuildTime AS (
  SELECT AVG(DATEDIFF(day, OrderDate, ShipDate)) AS OverallAvgTime
  FROM BIKE..Bicycle
)
SELECT YearlyBuildTime.Year, YearlyBuildTime.AvgBuildTime
FROM YearlyBuildTime, OverallAvgBuildTime
WHERE YearlyBuildTime.AvgBuildTime > OverallAvgBuildTime.OverallAvgTime;
